import re

import glob

import numpy as np

import matplotlib.pyplot as plt

from pathlib import Path

import pandas as pd

  

def parse_slurm_output(filename):

"""Parse SLURM output file to extract MD step data"""

data = []

# Pattern to match MD step lines: " 1 T= 1000. E= -.15252828E+04 F= -.15499716E+04"

pattern = r'^\s*(\d+)\s+T=\s*(\d+\.?\d*)\s+E=\s*([-\d.E+-]+)\s+F=\s*([-\d.E+-]+)'

with open(filename, 'r') as f:

for line in f:

match = re.match(pattern, line)

if match:

step = int(match.group(1))

temp = float(match.group(2))

energy = float(match.group(3))

free_energy = float(match.group(4))

data.append({

'step': step,

'temperature': temp,

'energy': energy,

'free_energy': free_energy

})

return pd.DataFrame(data)

  

def analyze_convergence():

"""Main analysis function"""

# Get all SLURM output files

slurm_files = glob.glob("slurm_CeO2_*.out")

if not slurm_files:

print("No SLURM output files found!")

return

print(f"Found {len(slurm_files)} SLURM output files")

# Parse all files

all_data = {}

for filename in slurm_files:

print(f"Parsing {filename}...")

# Extract temperature and block info from filename

# Example: slurm_CeO2_1000K_blk0_33887375.out

match = re.search(r'(\d+)K_blk(\d+)', filename)

if match:

temp = int(match.group(1))

block = int(match.group(2))

if temp not in all_data:

all_data[temp] = {}

df = parse_slurm_output(filename)

if not df.empty:

all_data[temp][block] = df

print(f" Temperature: {temp}K, Block: {block}, Steps: {len(df)}")

if not all_data:

print("No valid data found!")

return

# Create convergence plots

create_convergence_plots(all_data)

  

def create_convergence_plots(all_data):

"""Create comprehensive convergence plots"""

temps = sorted(all_data.keys())

n_temps = len(temps)

# Create figure with subplots

fig, axes = plt.subplots(n_temps, 2, figsize=(15, 4*n_temps))

if n_temps == 1:

axes = axes.reshape(1, -1)

colors = plt.cm.tab10(np.linspace(0, 1, 10))

for i, temp in enumerate(temps):

ax_temp, ax_energy = axes[i]

# Temperature convergence plot

for j, (block, df) in enumerate(all_data[temp].items()):

color = colors[j % len(colors)]

ax_temp.plot(df['step'], df['temperature'],

color=color, alpha=0.7, linewidth=1,

label=f'Block {block}')

ax_temp.axhline(temp, color='red', linestyle='--', linewidth=2, alpha=0.8, label='Target T')

ax_temp.set_title(f'{temp}K Temperature Convergence')

ax_temp.set_ylabel('Temperature (K)')

ax_temp.legend(fontsize=8)

ax_temp.grid(True, alpha=0.3)

# Energy convergence plot

for j, (block, df) in enumerate(all_data[temp].items()):

color = colors[j % len(colors)]

# Normalize energy to start at 0 for better comparison

normalized_energy = df['free_energy'] - df['free_energy'].iloc[0]

ax_energy.plot(df['step'], normalized_energy,

color=color, alpha=0.7, linewidth=1,

label=f'Block {block}')

ax_energy.set_title(f'{temp}K Energy Convergence')

ax_energy.set_ylabel('ΔF (eV)')

ax_energy.legend(fontsize=8)

ax_energy.grid(True, alpha=0.3)

# Set x-label for bottom row only

if i == n_temps - 1:

ax_temp.set_xlabel('MD Step')

ax_energy.set_xlabel('MD Step')

plt.tight_layout()

plt.savefig('convergence_analysis.png', dpi=300, bbox_inches='tight')

plt.show()

# Create summary statistics

create_summary_statistics(all_data)

  

def create_summary_statistics(all_data):

"""Create summary statistics table"""

print("\n" + "="*80)

print("CONVERGENCE SUMMARY STATISTICS")

print("="*80)

summary_data = []

for temp in sorted(all_data.keys()):

for block in sorted(all_data[temp].keys()):

df = all_data[temp][block]

# Calculate statistics

temp_mean = df['temperature'].mean()

temp_std = df['temperature'].std()

temp_target = temp

temp_error = abs(temp_mean - temp_target)

energy_start = df['free_energy'].iloc[0]

energy_end = df['free_energy'].iloc[-1]

energy_drift = energy_end - energy_start

# Temperature stability (how close to target)

temp_stability = 1.0 - (temp_error / temp_target)

summary_data.append({

'Temperature': f"{temp}K",

'Block': block,

'Steps': len(df),

'T_mean': f"{temp_mean:.1f}",

'T_std': f"{temp_std:.1f}",

'T_error': f"{temp_error:.1f}",

'T_stability': f"{temp_stability:.3f}",

'Energy_drift': f"{energy_drift:.4f}"

})

# Create summary table

df_summary = pd.DataFrame(summary_data)

print(df_summary.to_string(index=False))

# Save summary to file

df_summary.to_csv('convergence_summary.csv', index=False)

print(f"\nSummary saved to 'convergence_summary.csv'")

# Overall assessment

print("\n" + "="*80)

print("OVERALL CONVERGENCE ASSESSMENT")

print("="*80)

# Check temperature stability

temp_stabilities = [float(row['T_stability']) for row in summary_data]

avg_temp_stability = np.mean(temp_stabilities)

if avg_temp_stability > 0.95:

temp_status = "EXCELLENT"

elif avg_temp_stability > 0.90:

temp_status = "GOOD"

elif avg_temp_stability > 0.80:

temp_status = "FAIR"

else:

temp_status = "POOR"

print(f"Temperature Stability: {avg_temp_stability:.3f} ({temp_status})")

# Check energy convergence

energy_drifts = [abs(float(row['Energy_drift'])) for row in summary_data]

avg_energy_drift = np.mean(energy_drifts)

if avg_energy_drift < 0.01:

energy_status = "EXCELLENT"

elif avg_energy_drift < 0.05:

energy_status = "GOOD"

elif avg_energy_drift < 0.1:

energy_status = "FAIR"

else:

energy_status = "POOR"

print(f"Energy Convergence: {avg_energy_drift:.4f} eV ({energy_status})")

# Recommendations

print("\nRECOMMENDATIONS:")

if temp_status == "POOR":

print("- Check thermostat settings and time step")

if energy_status == "POOR":

print("- Consider longer equilibration time")

print("- Check for structural instabilities")

if temp_status in ["EXCELLENT", "GOOD"] and energy_status in ["EXCELLENT", "GOOD"]:

print("- Simulations appear well-converged")

print("- Consider extending production runs for better statistics")

  

if __name__ == "__main__":

analyze_convergence()
